3. Бірінші компонент

import { useState } from 'react';

function Counter() {
const [count, setCount] = useState(0);
return (
<div>
<h1>Сан: {count}</h1>
<button onClick={() => setCount(c => c + 1)}>+1</button>
</div>
);
}
