C

Басты бетке оралу

1. C тілі деген не? Неге үйренеміз?

C – ең жылдам, ең төменгі деңгейлі, ең күшті тілдердің бірі. Операциялық жүйелер, драйверлер, ядролар, ойындар, микроконтроллерлер – бәрі C-та жазылған!

2. Орнату және алғашқы программа

# Linux/macOS sudo apt install gcc # Ubuntu/Debian brew install gcc # macOS # Windows → MinGW немесе MSYS2 орнату
# hello.c #include int main() { printf("Сәлем, C әлемі!\n"); return 0; }
# Компиляция және іске қосу gcc hello.c -o hello ./hello
Сәлем, C әлемі!

3. Айнымалылар және деректер түрлері

int age = 25; // бүтін сан float height = 1.75; // ондық сан double pi = 3.14159; // дәлірек ондық char letter = 'A'; // бір символ char name[] = "Азамат"; // мәтін (массив)

4. printf() және scanf() – кіріс/шығыс

printf("Атыңыз: %s, жасыңыз: %d\n", name, age); scanf("%d", &age); // & – адрес операторы (маңызды!)

5. Шарттар (if, else)

if (age >= 18) { printf("Ересексіз!\n"); } else if (age >= 13) { printf("Жасөспірімсіз\n"); } else { printf("Баласыз\n"); }

6. Циклдар

// for for (int i = 1; i <= 10; i++) { printf("%d ", i); } // while int i = 0; while (i < 5) { printf("Сәлем %d\n", i++); } // do-while (кемінде бір рет орындайды) do { printf("Бұл шығады!\n"); } while (0);

7. Массивтер

int numbers[5] = {10, 20, 30, 40, 50}; for (int i = 0; i < 5; i++) { printf("numbers[%d] = %d\n", i, numbers[i]); } // Екі өлшемді массив int matrix[2][3] = { {1, 2, 3}, {4, 5, 6} };

8. Функциялар

// Прототип (алдын ала жариялау) int add(int a, int b); int main() { int sum = add(5, 7); printf("Қосынды: %d\n", sum); return 0; } // Анықтама int add(int a, int b) { return a + b; }

9. Pointer (Көрсеткіш) – C-тың жүрегі

int x = 10; int *p = &x; // p – x-тың адресі printf("x мәні: %d\n", x); printf("p арқылы: %d\n", *p); // *p = x мәні *p = 99; // x енді 99 болады! printf("Енді x = %d\n", x);

10. Структуры (struct)

typedef struct { char name[50]; int age; float gpa; } Student; Student s1; strcpy(s1.name, "Азамат"); s1.age = 20; s1.gpa = 3.8; printf("Студент: %s, GPA: %.2f\n", s1.name, s1.gpa);

11. Файлдармен жұмыс

FILE *file = fopen("data.txt", "w"); fprintf(file, "Сәлем файл!\n"); fclose(file); FILE *file2 = fopen("data.txt", "r"); char buffer[100]; fgets(buffer, 100, file2); printf("Файлдан: %s", buffer); fclose(file2);

12. Dynamic memory (malloc, free)

#include int *arr = (int*)malloc(5 * sizeof(int)); for (int i = 0; i < 5; i++) { arr[i] = i * 10; } free(arr); // міндетті түрде босату!

13. Senior деңгейі – нағыз C профи

// Bitwise операторлар x & y // AND x | y // OR x ^ y // XOR x << 2 // солға жылжыту (×4) x >> 1 // оңға жылжыту (÷2) // Preprocessor директивасы #define PI 3.14159 #define MAX(a,b) ((a)>(b)?(a):(b)) // Linked List (ең классика) typedef struct Node { int data; struct Node* next; } Node;
