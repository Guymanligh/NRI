2. Негізгі командалар

CREATE TABLE users (
id SERIAL PRIMARY KEY,
name VARCHAR(100) NOT NULL,
age INT CHECK (age >= 0)
);

INSERT INTO users (name, age) VALUES ('Азамат', 27);
UPDATE users SET age = 28 WHERE id = 1;
DELETE FROM users WHERE age < 18;

SELECT * FROM users WHERE age > 18 ORDER BY age DESC LIMIT 10;

-- JOIN
SELECT u.name, o.amount FROM orders o JOIN users u ON o.user_id = u.id;

3. Индекс, View, Trigger, Transaction

CREATE INDEX idx_age ON users(age);
CREATE VIEW adults AS SELECT * FROM users WHERE age >= 18;
